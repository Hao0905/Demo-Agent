## 8. Use case mẫu: Website Builder với AI

### 8.1 Mô tả

Xây dựng ứng dụng tạo website bằng AI. Luồng nghiệp vụ:

```
User nhập thông tin website
        │
        v

Trích xuất màu từ Logo (AI)
        │
        v

Tạo Website Profile + Theme
        │
        v

User chọn Section → nhập Prompt
        │
        v

AI chỉnh sửa nội dung Section
        │
        v

Render website với nội dung mới
```

### 8.4 Frontend Integration

```typescript
// ====== 1. GỬI THÔNG TIN WEBSITE ======

interface WebsiteProfile {
  companyName: string;
  industry: string;
  description: string;
  address: string;
  phone: string;
  email: string;
  style: "modern" | "professional" | "luxury";
}

async function createWebsiteProfile(profile: WebsiteProfile) {
  // Upload logo trước
  const logoFile = await uploadFile(profile.logoImage);

  // Gửi task với thông tin profile
  return streamTask(
    websiteBuilderDeploymentId,
    userId,
    `Tạo website profile mới:
    - Công ty: ${profile.companyName}
    - Lĩnh vực: ${profile.industry}
    - Mô tả: ${profile.description}
    - Style: ${profile.style}`,
    { parts: [{ kind: "file", file: { file_id: logoFile.file_id } }] },
    handleSSEEvent,
    handleError,
  );
}

// ====== 2. XỬ LÝ THEME SUGGESTIONS ======

interface ThemeSuggestion {
  name: string;
  primary: string;
  secondary: string;
  accent: string;
  text: string;
  border: string;
}

function handleSSEEvent({ event, data }: SSECallback) {
  switch (event) {
    case "artifact_update":
      // Agent trả về theme suggestions qua artifact
      if (data.last_chunk) {
        const suggestions: ThemeSuggestion[] = JSON.parse(data.content);
        showColorPicker(suggestions);
      }
      break;

    case "token":
      appendAssistantMessage(data.delta);
      break;

    case "final":
      // Task hoàn thành
      onProfileCreated(data.task);
      break;
  }
}

// ====== 3. CHỈNH SỬA SECTION BẰNG PROMPT ======

interface SectionEditRequest {
  sectionId: string;
  prompt: string;
  websiteProfile: WebsiteProfile;
  currentContent: string;
}

async function editSection(request: SectionEditRequest) {
  const contextMessage = `
    WEBSITE PROFILE:
    - Company: ${request.websiteProfile.companyName}
    - Industry: ${request.websiteProfile.industry}
    - Description: ${request.websiteProfile.description}
    - Style: ${request.websiteProfile.style}

    SECTION: ${request.sectionId}
    CURRENT CONTENT: ${request.currentContent}

    USER REQUEST: ${request.prompt}

    Hãy chỉnh sửa section này dựa trên thông tin trên.
    Chỉ thay đổi text và hình ảnh, giữ nguyên bố cục.
  `;

  return streamTask(
    websiteBuilderDeploymentId,
    userId,
    contextMessage,
    (evt) => {
      if (evt.event === "artifact_update" && evt.data.last_chunk) {
        const sectionData = JSON.parse(evt.data.content);
        renderSection(sectionData); // Render section đã chỉnh sửa
      }
    },
    handleError,
  );
}

// ====== 4. RENDER SECTION TRONG UI ======

function renderSection(data: {
  section_id: string;
  heading: string;
  content: string;
  images: string[];
  style: { alignment: string; background_color: string };
}) {
  return `
    <section id="${data.section_id}"
      style="text-align: ${data.style.alignment};
             background-color: ${data.style.background_color}">
      <h2>${data.heading}</h2>
      <div>${data.content}</div>
      ${data.images.map((src) => `<img src="${src}" />`).join("")}
    </section>
  `;
}
```

### 8.6 Xử lý nâng cao

**Lưu trữ session state:** Dùng `context_id` để group các task liên quan:

```typescript
// Task đầu tiên tạo context

const response = await fetch("/api/task/stream", {
  method: "POST",
  headers,
  body: JSON.stringify({
    deployment_id: websiteBuilderDeploymentId,
    user_id: userId,
    task_message: "Tạo website cho công ty ABC",
  }),
});
// Lấy context_id từ response

// Các task sau dùng chung context_id

fetch("/api/task/stream", {
  method: "POST",
  headers,
  body: JSON.stringify({
    deployment_id: websiteBuilderDeploymentId,
    user_id: userId,
    task_message: "Edit hero section",
    context_id: savedContextId, // 👈 Group các task
  }),
});
```

**Kiểm tra task status:**

```typescript
// Poll thay vì stream

const response = await fetch("/api/task/send", { ... });
const { task_id } = await response.json();

// Poll cho đến khi hoàn thành

const interval = setInterval(async () => {
  const info = await fetch(
    `/api/task/info/${task_id}?deployment_id=${deploymentId}`,
    { headers }
  );
  const task = await info.json();
  if (task.state === "completed" || task.state === "failed" || task.state === "canceled") {
    clearInterval(interval);
    processResult(task);
  }
}, 2000);
```

**Cancel task:**

```typescript

await fetch(
  `/api/task/cancel?task_id=${taskId}&deployment_id=${deploymentId}`,
  {
    method: "POST",
    headers,
  },
);
```
