### 7.4 Gửi Task với Streaming (SSE)

Dùng cho real-time UX (chat, progress bar, typewriter effect):

```typescript

type SSECallback = {
  event: string;
  data: Record<string, unknown>;
};

function streamTask(
  deploymentId: string,
  userId: string,
  prompt: string,
  onEvent: (evt: SSECallback) => void,
  onError?: (err: Error) => void,
): () => void {
  const controller = new AbortController();

  fetch("/api/task/stream", {
    method: "POST",
    headers: {
      "Content-Type": "application/json",
      "X-API-Key": getApiKey(),
    },
    body: JSON.stringify({
      deployment_id: deploymentId,
      user_id: userId,
      task_message: prompt,
    }),
    signal: controller.signal,
  })
    .then(async (response) => {
      if (!response.ok) {
        const text = await response.text();
        throw new Error(text || `HTTP ${response.status}`);
      }
      const reader = response.body!.getReader();
      const decoder = new TextDecoder();
      let buffer = "";

      while (true) {
        const { done, value } = await reader.read();
        if (done) break;

        buffer += decoder.decode(value, { stream: true });
        const lines = buffer.split("\n\n");
        buffer = lines.pop() ?? "";

        for (const chunk of lines) {
          const eventLine = chunk
            .split("\n")
            .find((l) => l.startsWith("event:"));
          const dataLine = chunk.split("\n").find((l) => l.startsWith("data:"));
          if (!eventLine || !dataLine) continue;

          const event = eventLine.replace("event:", "").trim();
          const data = JSON.parse(dataLine.replace("data:", "").trim());
          onEvent({ event, data });
        }
      }
    })
    .catch((err) => {
      if (err.name !== "AbortError" && onError) onError(err);
    });

  return () => controller.abort(); // Cleanup: gọi khi component unmount
}
```

### 7.5 Xử lý các SSE event types

```typescript

streamTask(deploymentId, userId, prompt, ({ event, data }) => {
  switch (event) {
    case "token":
      // Text delta — append vào message đang hiển thị
      appendToMessage(data.delta);
      break;

    case "thinking":
      // ADK reasoning trace — hiển thị ở panel thinking
      showThinking(data.author, data.delta);
      break;

    case "artifact_update":
      // Agent output artifact — render khi last_chunk = true
      if (data.last_chunk) {
        renderArtifact(data.content);
      }
      break;

    case "status_update":
      // Task state transition
      updateStatus(data.state, data.final);
      break;

    case "input-required":
    case "auth-required":
      // Task cần input thêm — gọi continue
      showInputRequest(data.message);
      break;

    case "error":
      showError(data.code, data.message);
      break;

    case "final":
      // Stream kết thúc — task hoàn thành
      onTaskComplete(data.content, data.task);
      break;
  }
});
```

### 7.7 Upload file

```typescript

async function uploadFile(file: File): Promise<{ file_id: string }> {
  const formData = new FormData();
  formData.append("file", file);

  const response = await fetch("/api/files/upload", {
    method: "POST",
    headers: { "X-API-Key": getApiKey() }, // Không set Content-Type — browser tự set với boundary
    body: formData,
  });
  return response.json();
}

// Sau đó dùng file_id trong task message
```
